```rust
#[derive(Debug, Clone)]
pub struct ShadowRebalanceRow {
    pub pool_address: String,
    pub trigger_reason: String, // 'out_of_range' | 'il_threshold' | 'manual'
    pub price: f64,
    pub simulated_range_width: Option<f64>,
    pub simulated_fees_earned: Option<f64>,
    pub simulated_il_usd: Option<f64>,
    pub simulated_net_pnl: Option<f64>,
    pub error_flag: bool,
    pub error_message: Option<String>,
}

pub async fn write_shadow_rebalance(
    pool: &sqlx::PgPool,
    row: &ShadowRebalanceRow,
) -> anyhow::Result<()> {
    sqlx::query(
        r#"
        INSERT INTO shadow_rebalances (
            pool_address, trigger_reason, price,
            simulated_range_width, simulated_fees_earned,
            simulated_il_usd, simulated_net_pnl,
            error_flag, error_message
        ) VALUES ($1,$2,$3,$4,$5,$6,$7,$8,$9)
        "#,
    )
    .bind(&row.pool_address)
    .bind(&row.trigger_reason)
    .bind(row.price)
    .bind(row.simulated_range_width)
    .bind(row.simulated_fees_earned)
    .bind(row.simulated_il_usd)
    .bind(row.simulated_net_pnl)
    .bind(row.error_flag)
    .bind(row.error_message.as_deref())
    .execute(pool)
    .await?;
    Ok(())
}

/// Fire-and-forget spawn, mirroring spawn_pnl_write.
pub fn spawn_shadow_write(pool: sqlx::PgPool, row: ShadowRebalanceRow) {
    tokio::spawn(async move {
        if let Err(e) = write_shadow_rebalance(&pool, &row).await {
            tracing::error!(error = %e, "failed to write shadow_rebalances row");
        }
    });
}
```

```rust
let decision_result: anyhow::Result<Option<RebalanceDecision>> = (|| {
    let signal = strategy::should_rebalance(&pool_state, &position, &config)?;
    if !signal.should_rebalance {
        return Ok(None);
    }
    let plan = execution::build_rebalance_plan(&pool_state, &position, &signal)?;
    Ok(Some(RebalanceDecision { signal, plan }))
})();

let row = match &decision_result {
    Ok(Some(dec)) => ShadowRebalanceRow {
        pool_address: pool_address.to_string(),
        trigger_reason: dec.signal.trigger_reason.to_string(), // "out_of_range" | "il_threshold" | "manual"
        price: current_price,
        simulated_range_width: Some(dec.plan.new_range_width()),
        simulated_fees_earned: Some(dec.plan.simulated_fees_earned),
        simulated_il_usd: Some(dec.plan.simulated_il_usd),
        simulated_net_pnl: Some(dec.plan.simulated_fees_earned - dec.plan.simulated_il_usd.abs()),
        error_flag: false,
        error_message: None,
    },
    Ok(None) => {
        // No rebalance needed this tick — do not write a row.
        // (We only persist decisions, not non-decisions.)
        return_early_skip_shadow_write = true;
        /* sentinel, skip write */
        ShadowRebalanceRow { pool_address: String::new(), trigger_reason: String::new(), price: 0.0,
            simulated_range_width: None, simulated_fees_earned: None, simulated_il_usd: None,
            simulated_net_pnl: None, error_flag: false, error_message: None }
    },
    Err(e) => ShadowRebalanceRow {
        pool_address: pool_address.to_string(),
        trigger_reason: "error".to_string(),
        price: current_price,
        simulated_range_width: None,
        simulated_fees_earned: None,
        simulated_il_usd: None,
        simulated_net_pnl: None,
        error_flag: true,
        error_message: Some(format!("{:#}", e)),
    },
};

if !matches!(&decision_result, Ok(None)) {
    tracing::info!(
        pool = %row.pool_address,
        trigger = %row.trigger_reason,
        price = row.price,
```
