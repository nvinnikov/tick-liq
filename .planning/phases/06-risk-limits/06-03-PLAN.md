```rust
// --- Risk gate (D-04: every tick; D-05: after pnl_write, before should_rebalance) ---

// Fetch Drift margin ratio via synchronous RPC (blocking, with 5s timeout)
// Use spawn_blocking to avoid blocking the event loop
let drift_margin = if drift_min_margin_ratio_val.is_some() {
    // Clone what we need for the blocking call
    let rm_rpc_url = rpc_url.clone();
    let rm_pubkey = drift_user_pubkey;
    let rm_threshold = drift_min_margin_ratio_val;
    tokio::task::spawn_blocking(move || {
        // Create a temporary monitor just for the fetch, or call the static method
        // The fetch_drift_margin_ratio on the risk_monitor handles None pubkey internally
        risk_monitor.fetch_drift_margin_ratio()
    }).await.unwrap_or(None)
} else {
    None
};

let risk_action = risk_monitor.evaluate(&snap, drift_margin);

match &risk_action {
    RiskAction::HaltAll { drawdown_pct } => {
        tracing::error!(drawdown_pct, "risk: drawdown limit breached -- halting");
        // Close LP position via OrcaExecutor (D-10)
        // Only attempt if in live mode and executor is available
        if let (Some(ref executor), strategy::signal::RebalanceDecision::Rebalance { .. } | _) =
            (&orca_executor_opt, strategy::signal::RebalanceDecision::Hold { reason: String::new() })
        {
            // Attempt LP close -- per D-10: close_position + collect_fees
            match executor.execute_close_position(/* position params */).await {
                Ok(sig) => tracing::info!(%sig, "risk: LP position closed after drawdown breach"),
                Err(e) => tracing::error!(error = %e, "risk: LP close failed after drawdown breach"),
            }
        }
        // Log Drift hedge close as deferred (D-10)
        tracing::error!("halt: drawdown limit hit -- Drift hedge close deferred (LIVE-02)");
        // Persist halt_flag to DB (D-11)
        RiskMonitor::persist_state(pg.clone(), risk_monitor.state.clone());
        continue; // Skip rest of tick cycle (D-06)
    }
    RiskAction::PauseRebalancing { il_pct } => {
        tracing::warn!(il_pct, "risk: IL pause active -- skipping rebalance");
        RiskMonitor::persist_state(pg.clone(), risk_monitor.state.clone());
        continue; // Skip should_rebalance (D-06)
    }
    RiskAction::ResumeRebalancing { il_pct } => {
        tracing::info!(il_pct, "risk: IL recovered -- resuming rebalance");
        RiskMonitor::persist_state(pg.clone(), risk_monitor.state.clone());
        // Fall through to should_rebalance()
    }
    RiskAction::CloseDriftHedge { margin_ratio } => {
        tracing::error!(
            margin_ratio,
            "risk: Drift margin below threshold -- hedge close deferred (LIVE-02)"
        );
        RiskMonitor::persist_state(pg.clone(), risk_monitor.state.clone());
        // LP rebalance continues (RISK-03: only Drift side affected)
        // Fall through to should_rebalance()
    }
    RiskAction::Continue => {
        // Persist state (peak_pnl may have been updated)
        RiskMonitor::persist_state(pg.clone(), risk_monitor.state.clone());
        // Fall through to should_rebalance()
    }
}
```
