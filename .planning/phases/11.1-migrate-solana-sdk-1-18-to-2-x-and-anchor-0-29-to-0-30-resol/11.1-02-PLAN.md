### 2.1 Шаблон нового файла

```rust
use std::sync::{Arc, RwLock};
use std::time::Instant;
use tokio::sync::broadcast;
use tracing::{info, warn};

use binance_sdk::config::ConfigurationWebsocketStreams;
use binance_sdk::spot::SpotWsStreams;
use binance_sdk::spot::websocket_streams::{BookTickerParams, BookTickerResponse};

pub struct CexPrice {
    pub price: f64,
    pub updated_at: Instant,
}

pub type CexPriceState = Arc<RwLock<Option<CexPrice>>>;

/// Apply a BookTicker update to shared state. Returns true if state was written,
/// false if the incoming data failed to parse (logs a warn! in that case).
///
/// Extracted for unit testing — replaces the old `handle_frame`.
fn apply_book_ticker(
    best_bid_price: &str,
    best_ask_price: &str,
    state: &CexPriceState,
) -> bool {
    let bid: f64 = match best_bid_price.parse() {
        Ok(v) => v,
        Err(_) => {
            warn!("cex_ws: non-numeric bid '{}', skipping", best_bid_price);
            return false;
        }
    };
    let ask: f64 = match best_ask_price.parse() {
        Ok(v) => v,
        Err(_) => {
            warn!("cex_ws: non-numeric ask '{}', skipping", best_ask_price);
            return false;
        }
    };
    let mid = (bid + ask) / 2.0;
    let mut guard = state.write().unwrap_or_else(|p| p.into_inner());
    *guard = Some(CexPrice { price: mid, updated_at: Instant::now() });
    true
}

/// Watch Binance bookTicker stream for `symbol` (case-insensitive; lowercased by SDK params).
/// Writes mid-price = (bid + ask) / 2 into `state` on every update.
/// Reconnect / ping-pong / backoff handled by binance-sdk v45.
/// Returns on shutdown broadcast.
pub async fn watch_binance_price(
    symbol: String,
    state: CexPriceState,
    mut shutdown: broadcast::Receiver<()>,
) {
    let cfg = match ConfigurationWebsocketStreams::builder().build() {
        Ok(c) => c,
        Err(e) => {
            warn!("cex_ws: config build error: {}", e);
            return;
        }
    };
    let client = SpotWsStreams::production(cfg);
    let connection = match client.connect().await {
        Ok(c) => c,
        Err(e) => {
            warn!("cex_ws: connect error: {}", e);
            return;
        }
    };
    let params = match BookTickerParams::builder(symbol.to_lowercase()).build() {
        Ok(p) => p,
        Err(e) => {
            warn!("cex_ws: params build error: {}", e);
            let _ = connection.disconnect().await;
            return;
        }
    };
    let stream = match connection.book_ticker(params).await {
        Ok(s) => s,
        Err(e) => {
            warn!("cex_ws: book_ticker subscribe error: {}", e);
            let _ = connection.disconnect().await;
            return;
        }
    };
    info!("cex_ws: connected, subscribed to bookTicker");

    let state_cb = state.clone();
    stream.on_message(move |data: BookTickerResponse| {
        // UNVERIFIED — field names `best_bid_price` / `best_ask_price` assumed
        // (A5 in RESEARCH). Verify via `cargo doc --open -p binance-sdk` BEFORE
        // finalizing this closure (OQ-4 spike — 15 minutes max).
        apply_book_ticker(&data.best_bid_price, &data.best_ask_price, &state_cb);
    });

    // Block until shutdown, then gracefully disconnect.
    let _ = shutdown.recv().await;
    info!("cex_ws: shutdown received, disconnecting");
    if let Err(e) = connection.disconnect().await {
        warn!("cex_ws: disconnect error: {}", e);
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn cex_price_struct_has_public_fields() {
        let p = CexPrice {
            price: 10.0,
            updated_at: Instant::now(),
        };
        assert_eq!(p.price, 10.0);
        let _ = p.updated_at;
    }

    #[test]
    fn apply_book_ticker_updates_state() {
        let state: CexPriceState = Arc::new(RwLock::new(None));
        let updated = apply_book_ticker("100.0", "102.0", &state);
        assert!(updated);
        let guard = state.read().expect("read");
        let cp = guard.as_ref().expect("some");
        assert!((cp.price - 101.0).abs() < 1e-9);
    }

    #[test]
    fn apply_book_ticker_ignores_malformed_and_keeps_state() {
        let state: CexPriceState = Arc::new(RwLock::new(None));
        let updated = apply_book_ticker("not-a-number", "1.0", &state);
        assert!(!updated);
        assert!(state.read().expect("read").is_none());

        let updated2 = apply_book_ticker("1.0", "nope", &state);
        assert!(!updated2);
        assert!(state.read().expect("read").is_none());
    }
}
```

### 2.3 Field-type branch (расширено per checker review)

В зависимости от того, что покажет spike:

**Сценарий happy-path (A5 VERIFIED):** поля `best_bid_price: String` и `best_ask_price: String` как предполагает шаблон выше. Никаких правок signature / tests — всё уже написано.

**Сценарий field types differ (f64, Option<String>, другие имена):**
Это ЛЕГИТИМНАЯ in-task модификация — НЕ CHECKPOINT trigger. Executor вносит все правки в тот же коммит Task 3:

1. Обновить сигнатуру `apply_book_ticker`:
   - Если `best_bid_price: f64` — изменить тип параметров на `f64`, убрать `.parse()` внутри (напрямую использовать); вернуть false если NaN или ≤0.
   - Если `best_bid_price: Option<String>` — обновить сигнатуру на `Option<&str>` либо внутри `.as_deref().unwrap_or("")`; сохранить behaviour "ignore and keep state" на `None`.
   - Если имена полей `bidPrice`/`askPrice` или `b`/`a` — в closure использовать реальные имена (`&data.<actual>`), signature `apply_book_ticker` может остаться `&str`.

2. Обновить ОБА unit-теста (`apply_book_ticker_updates_state`, `apply_book_ticker_ignores_malformed_and_keeps_state`) под новую сигнатуру. Сохранить поведенческие инварианты:
   - valid input → `returns true` + state становится `Some` с корректным mid-price (проверка `(p.price - 101.0).abs() < 1e-9` эквивалент).
   - malformed input (non-numeric OR NaN OR None OR negative) → `returns false` + state остаётся `None` (или предыдущим значением, если уже было Some).

3. Closure в `stream.on_message` передаёт ровно те типы, что принимает `apply_book_ticker` после правок.

**CHECKPOINT trigger (только):** если API surface binance-sdk требует контрактной смены нашей публичной функции `watch_binance_price`:
   - Сигнатура `async fn` → sync closure вместо await (невозможно без рефакторинга main.rs).
   - Другие error types / lifetimes, ломающие `Send + 'static` bound (влечёт рефакторинг spawn'а в main.rs).
   - SDK требует явный runtime init вне `tokio::spawn`.
Любое из вышеперечисленного — BLOCKER, поднять CHECKPOINT. Ограниченные within-adapter изменения (типы параметров `apply_book_ticker`, имена полей в closure) CHECKPOINT НЕ требуют.

Spike + правки field types ограничены 15 минутами (D-OQ4); если после 15 минут поля не найдены (doc недоступен, SDK signature нечитаема) — поднять CHECKPOINT с выводом `cargo doc` и ссылкой на docs.rs.
