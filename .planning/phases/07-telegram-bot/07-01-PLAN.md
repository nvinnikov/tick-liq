3. **src/bot/mod.rs**: Create the bot module root.
```rust
pub mod commands;

use anyhow::Result;
use sqlx_postgres::PgPool;
use std::sync::{Arc, Mutex};
use teloxide::prelude::*;
use tokio::sync::oneshot;

use crate::strategy::risk_monitor::RiskMonitor;

/// Shared state passed to every bot command handler.
#[derive(Clone)]
pub struct BotState {
    pub db_pool: PgPool,
    pub risk_monitor: Arc<Mutex<RiskMonitor>>,
    pub pool_address: String,
    pub mint: String,
    /// When a rebalance proposal is pending, holds the oneshot sender.
    /// `/approve` sends `true`, timeout or `/reject` sends nothing (sender dropped).
    pub pending_approval: Arc<Mutex<Option<oneshot::Sender<bool>>>>,
}

/// Spawn the Telegram bot as a background tokio task.
///
/// Returns a `JoinHandle` so the watch loop can detect bot crashes.
/// Reads `TELEGRAM_BOT_TOKEN` from environment (panics if absent per TG security).
pub async fn spawn_bot(state: BotState) -> Result<tokio::task::JoinHandle<()>> {
    let token = std::env::var("TELEGRAM_BOT_TOKEN")
        .map_err(|_| anyhow::anyhow!("TELEGRAM_BOT_TOKEN env var is required for bot mode"))?;

    let bot = Bot::new(token);

    let handler = commands::build_handler();

    let handle = tokio::spawn(async move {
        Dispatcher::builder(bot, handler)
            .dependencies(dptree::deps![state])
            .enable_ctrlc_handler()
            .build()
            .dispatch()
            .await;
    });

    Ok(handle)
}
```

use teloxide::{prelude::*, utils::command::BotCommands, dispatching::UpdateHandler};
use dptree::case;

use super::BotState;

#[derive(BotCommands, Clone)]
#[command(rename_rule = "lowercase", description = "LP Manager commands:")]
pub enum Command {
    #[command(description = "Show current position status and risk metrics")]
    Status,
    #[command(description = "Pause rebalancing (position stays open)")]
    Pause,
    #[command(description = "Resume rebalancing")]
    Resume,
    #[command(description = "24h P&L report")]
    Report,
    #[command(description = "Approve pending rebalance")]
    Approve,
}

pub fn build_handler() -> UpdateHandler<anyhow::Error> {
    let command_handler = teloxide::filter_command::<Command, _>()
        .branch(case![Command::Status].endpoint(handle_status))
        .branch(case![Command::Pause].endpoint(handle_pause))
        .branch(case![Command::Resume].endpoint(handle_resume))
        .branch(case![Command::Report].endpoint(handle_report))
        .branch(case![Command::Approve].endpoint(handle_approve));

    Update::filter_message()
        .branch(command_handler)
}

<threat_model>
## Assets
- TELEGRAM_BOT_TOKEN: Telegram bot API credential (medium sensitivity -- allows sending messages as the bot)
- Operator approval flow: unauthorized /approve could trigger unwanted rebalances (high impact)

## Threat Actors
- Network observer: could intercept bot token if transmitted insecurely (low -- Telegram API uses HTTPS)
- Unauthorized Telegram user: could send /approve to the bot (medium)

## Attack Vectors
- T-07-01: Bot token in config file or CLI arg (medium likelihood / high impact) -> Mitigation: Token loaded exclusively from TELEGRAM_BOT_TOKEN env var via std::env::var (never in config/CLI). Task 1 enforces this.
- T-07-02: Unauthorized user sends /approve (medium likelihood / high impact) -> Mitigation: Deferred to Plan 02 where chat_id allowlist is implemented. Noted as residual risk for this plan.
- T-07-03: SQL injection via bot command input (low likelihood / high impact) -> Mitigation: All DB queries use parameterized sqlx::query binds, no string interpolation.

## Residual Risks
- T-07-02 (chat_id filtering) accepted for Plan 01 -- handler stubs do not execute real actions. Full mitigation in Plan 02/03.
</threat_model>
