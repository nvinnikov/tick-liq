use anyhow::{bail, Result};
use chrono::{DateTime, Datelike, NaiveDate, Utc};

use crate::backtest::{BacktestParams, BacktestResult, DayResult, price_to_tick};
use crate::math::il::compute_il;
use crate::storage::tick_reader::PoolTickRow;
use crate::strategy;

/// Backtest input for DB mode. Unlike GBM mode, entry_price/range are derived
/// from the first tick + the tick_spacing/range_width policy in BacktestParams.
pub struct DbBacktestInput {
    pub params: BacktestParams,
    pub position_liquidity: u128, // derived from --capital in CLI layer (plan 03-03)
    pub price_lower: f64,
    pub price_upper: f64,
    pub entry_price: f64,
}

/// Convert sqrt_price (u128, X64 fixed-point) to floating price.
/// Matches the watch-loop derivation: price = (sqrt_price / 2^64)^2.
#[inline]
fn sqrt_price_to_price(sqrt_price: u128) -> f64 {
    let s = sqrt_price as f64 / (u128::pow(2, 64) as f64);
    s * s
}

/// Compute fee-growth delta with u128 wrapping support.
#[inline]
fn fee_growth_delta(prev: u128, curr: u128) -> u128 {
    curr.wrapping_sub(prev)
}

/// Replay a chronological slice of PoolTickRow and produce a BacktestResult
/// with the same schema as the GBM `backtest::run()`.
///
/// NOTE: Fee accrual uses the approximation
///     fees_step = (fee_growth_delta_a * position_liquidity / pool_liquidity)
/// scaled out of X64 fixed-point. Exact fee_growth_inside requires per-position
/// tick array data not stored in `pool_ticks` — documented approximation.
pub fn run_db_backtest(
    input: DbBacktestInput,
    ticks: &[PoolTickRow],
) -> Result<BacktestResult> {
    if ticks.is_empty() {
        bail!("run_db_backtest: empty tick stream");
    }

    let tick_spacing = input.params.tick_spacing;
    let mut tick_lower = price_to_tick(input.price_lower, tick_spacing);
    let mut tick_upper = price_to_tick(input.price_upper, tick_spacing);

    let mut cumulative_fees: f64 = 0.0;
    let mut rebalance_count: u32 = 0;
    let mut prev_fg_a: u128 = ticks[0].fee_growth_global_a;

    let mut days: Vec<DayResult> = Vec::new();
    let mut current_day: Option<NaiveDate> = None;
    let mut day_fees_start: f64 = 0.0;

    for (i, t) in ticks.iter().enumerate() {
        let price = sqrt_price_to_price(t.sqrt_price);
        let day = t.time.date_naive();

        // Start new day roll-up when date changes
        if current_day.map(|d| d != day).unwrap_or(true) {
            if let Some(prev_day) = current_day {
                let day_fees = cumulative_fees - day_fees_start;
                let il = compute_il(input.entry_price, price, input.price_lower, input.price_upper);
                let position_value = input.params.capital + cumulative_fees + il;
                days.push(DayResult {
                    day: prev_day,
                    price,
                    fees_earned: day_fees,
                    il_usd: il,
                    net_pnl: cumulative_fees + il,
                    position_value,
                    rebalance_count,
                });
            }
            current_day = Some(day);
            day_fees_start = cumulative_fees;
        }

        // Fee accrual from delta, skip first tick (no prior reference)
        if i > 0 && t.liquidity > 0 {
            let delta = fee_growth_delta(prev_fg_a, t.fee_growth_global_a);
            // X64 fixed-point: divide by 2^64 and scale by liquidity share.
            let share = (input.position_liquidity as f64) / (t.liquidity as f64);
            let fees_raw = (delta as f64) / (u128::pow(2, 64) as f64);
            cumulative_fees += fees_raw * share;
        }
        prev_fg_a = t.fee_growth_global_a;

        // Rebalance signal per tick
        let il = compute_il(input.entry_price, price, input.price_lower, input.price_upper);
        let net_pnl = cumulative_fees + il;
        if input.params.rebalance
            && strategy::should_rebalance(t.tick_current, tick_lower, tick_upper, net_pnl, &input.params.strategy_cfg)
        {
            tick_lower = price_to_tick(price * input.params.range_factor_lower, tick_spacing);
            tick_upper = price_to_tick(price * input.params.range_factor_upper, tick_spacing);
            rebalance_count += 1;
        }
    }

    // Flush final day
    if let Some(prev_day) = current_day {
        let last = ticks.last().unwrap();
        let price = sqrt_price_to_price(last.sqrt_price);
        let day_fees = cumulative_fees - day_fees_start;
        let il = compute_il(input.entry_price, price, input.price_lower, input.price_upper);
        let position_value = input.params.capital + cumulative_fees + il;
        days.push(DayResult {
            day: prev_day,
            price,
            fees_earned: day_fees,
            il_usd: il,
            net_pnl: cumulative_fees + il,
            position_value,
            rebalance_count,
        });
    }

    let final_price = sqrt_price_to_price(ticks.last().unwrap().sqrt_price);
    let final_il = compute_il(input.entry_price, final_price, input.price_lower, input.price_upper);

    Ok(BacktestResult {
        days,
        total_fees: cumulative_fees,
        total_il: final_il,
        net_pnl: cumulative_fees + final_il,
        rebalance_count,
    })
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn fee_growth_delta_wraps() {
        assert_eq!(fee_growth_delta(u128::MAX, 5), 6);
        assert_eq!(fee_growth_delta(100, 200), 100);
    }

    #[test]
    fn sqrt_price_conversion_matches_watch_loop() {
        // sqrt_price = 2^64 → price = 1.0
        let p = sqrt_price_to_price(1u128 << 64);
        assert!((p - 1.0).abs() < 1e-9);
    }
}
