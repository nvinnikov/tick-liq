In the `Commands::Backtest` match arm, dispatch:

```rust
Commands::Backtest { pool, from, to, synthetic, capital, fee_bps, tick_spacing, rebalance, /* GBM-only fields */ .. } => {
    if synthetic || pool.is_none() {
        // Existing GBM path — call backtest::run(params) exactly as before
        backtest::run(/* existing param construction */)?;
    } else {
        // DB mode
        let pool_addr = pool.as_ref().unwrap();
        let from = from.ok_or_else(|| anyhow::anyhow!("--from is required in DB mode (YYYY-MM-DD)"))?;
        let to = to.ok_or_else(|| anyhow::anyhow!("--to is required in DB mode (YYYY-MM-DD)"))?;

        let pg = storage::connect().await?; // or whatever the existing PgPool constructor is called
        let ticks = storage::tick_reader::read_ticks(&pg, pool_addr, from, to).await?;

        if ticks.is_empty() {
            anyhow::bail!(
                "No pool_ticks data for pool {} between {} and {}. Run 'watch' to accumulate data, or use --synthetic for a GBM simulation.",
                pool_addr, from, to
            );
        }

        // Derive entry_price from first tick; price_lower/upper from capital + tick_spacing policy.
        let first_sqrt = ticks[0].sqrt_price as f64 / (u128::pow(2, 64) as f64);
        let entry_price = first_sqrt * first_sqrt;
        let price_lower = entry_price * 0.95;
        let price_upper = entry_price * 1.05;

        // Position liquidity: derive from --capital against the first tick's pool state.
        // Simple heuristic: position_liquidity = capital as u128 (USD → units),
        // refined in future phase. Documented as a known approximation.
        let position_liquidity: u128 = (capital as u128).max(1);

        let params = backtest::BacktestParams {
            capital,
            fee_bps,
            tick_spacing,
            rebalance,
            // ... any strategy_cfg / range_factors default construction ...
            ..Default::default()
        };

        let input = backtest::db_replay::DbBacktestInput {
            params,
            position_liquidity,
            price_lower,
            price_upper,
            entry_price,
        };

        let result = backtest::db_replay::run_db_backtest(input, &ticks)?;
        backtest::print_results(&result);
    }
}
```

```rust
use chrono::{TimeZone, Utc};
use tick_liq::backtest::db_replay::{run_db_backtest, DbBacktestInput};
use tick_liq::backtest::BacktestParams;
use tick_liq::storage::tick_reader::PoolTickRow;

fn mk_tick(
    ts: chrono::DateTime<Utc>,
    slot: i64,
    sqrt_price: u128,
    liquidity: u128,
    fga: u128,
) -> PoolTickRow {
    PoolTickRow {
        time: ts,
        pool_address: "TESTPOOL".into(),
        slot,
        tick_current: 0,
        sqrt_price,
        liquidity,
        fee_growth_global_a: fga,
        fee_growth_global_b: 0,
    }
}

#[test]
fn db_mode_fixture_backtest() {
    let one = 1u128 << 64; // sqrt_price = 1.0
    let ticks = vec![
        mk_tick(Utc.with_ymd_and_hms(2026, 1, 1, 12, 0, 0).unwrap(), 1, one, 1_000_000, 0),
        mk_tick(Utc.with_ymd_and_hms(2026, 1, 1, 13, 0, 0).unwrap(), 2, one, 1_000_000, 1_000_000),
        mk_tick(Utc.with_ymd_and_hms(2026, 1, 2, 12, 0, 0).unwrap(), 3, one, 1_000_000, 5_000_000),
    ];

    let input = DbBacktestInput {
        params: BacktestParams {
            capital: 10_000.0,
            fee_bps: 5,
            tick_spacing: 64,
            rebalance: false,
            ..Default::default()
        },
        position_liquidity: 100_000,
        price_lower: 0.95,
        price_upper: 1.05,
        entry_price: 1.0,
    };

    let result = run_db_backtest(input, &ticks).expect("replay ok");

    assert_eq!(result.days.len(), 2, "two distinct UTC days");
    assert!(result.total_fees >= 0.0, "fees non-negative");
    assert!(result.total_fees.is_finite(), "fees finite");
    assert!(result.net_pnl.is_finite(), "net_pnl finite");
}

#[test]
fn db_mode_handles_fee_growth_wrap() {
    let one = 1u128 << 64;
    let ticks = vec![
        mk_tick(Utc.with_ymd_and_hms(2026, 1, 1, 0, 0, 0).unwrap(), 1, one, 1_000_000, u128::MAX - 10),
        mk_tick(Utc.with_ymd_and_hms(2026, 1, 1, 1, 0, 0).unwrap(), 2, one, 1_000_000, 5),
    ];

    let input = DbBacktestInput {
        params: BacktestParams {
            capital: 10_000.0,
            fee_bps: 5,
            tick_spacing: 64,
            rebalance: false,
            ..Default::default()
        },
        position_liquidity: 100_000,
        price_lower: 0.95,
        price_upper: 1.05,
        entry_price: 1.0,
    };

    let result = run_db_backtest(input, &ticks).expect("wrap handled");
    assert!(result.total_fees.is_finite(), "no NaN/Inf on wrap");
    assert!(result.total_fees >= 0.0, "wrapped delta still yields non-negative fees");
}
```

If the crate name in Cargo.toml is not `tick_liq`, adjust `use` paths accordingly. If `BacktestParams` lacks `Default`, construct explicitly with every field set.
