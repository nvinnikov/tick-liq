```rust
use anyhow::{Context, Result};
use chrono::{DateTime, NaiveDate, Utc};
use sqlx::{PgPool, Row};

#[derive(Debug, Clone)]
pub struct PoolTickRow {
    pub time: DateTime<Utc>,
    pub pool_address: String,
    pub slot: i64,
    pub tick_current: i32,
    pub sqrt_price: u128,
    pub liquidity: u128,
    pub fee_growth_global_a: u128,
    pub fee_growth_global_b: u128,
}

pub async fn read_ticks(
    pool: &PgPool,
    pool_address: &str,
    from: NaiveDate,
    to: NaiveDate,
) -> Result<Vec<PoolTickRow>> {
    let from_ts = from.and_hms_opt(0, 0, 0).unwrap().and_utc();
    let to_ts = to.and_hms_opt(0, 0, 0).unwrap().and_utc();

    let rows = sqlx::query(
        "SELECT time, pool_address, slot, tick_current, \
         sqrt_price::text AS sqrt_price, \
         liquidity::text AS liquidity, \
         fee_growth_global_a::text AS fee_growth_global_a, \
         fee_growth_global_b::text AS fee_growth_global_b \
         FROM pool_ticks \
         WHERE pool_address = $1 AND time >= $2 AND time < $3 \
         ORDER BY time ASC, slot ASC",
    )
    .bind(pool_address)
    .bind(from_ts)
    .bind(to_ts)
    .fetch_all(pool)
    .await
    .context("query pool_ticks")?;

    let mut out = Vec::with_capacity(rows.len());
    for r in rows {
        let sqrt_price: String = r.try_get("sqrt_price")?;
        let liquidity: String = r.try_get("liquidity")?;
        let fga: String = r.try_get("fee_growth_global_a")?;
        let fgb: String = r.try_get("fee_growth_global_b")?;
        out.push(PoolTickRow {
            time: r.try_get("time")?,
            pool_address: r.try_get("pool_address")?,
            slot: r.try_get("slot")?,
            tick_current: r.try_get("tick_current")?,
            sqrt_price: sqrt_price.parse().context("parse sqrt_price u128")?,
            liquidity: liquidity.parse().context("parse liquidity u128")?,
            fee_growth_global_a: fga.parse().context("parse fee_growth_global_a u128")?,
            fee_growth_global_b: fgb.parse().context("parse fee_growth_global_b u128")?,
        });
    }
    Ok(out)
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI args → SQL | User-supplied pool_address string crosses into DB query |
| DB NUMERIC → u128 | Untrusted string conversion from DB into bounded Rust int |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering (SQLi) | read_ticks pool_address param | mitigate | Use `sqlx::query(...).bind(pool_address)` parameterized binding — no string concatenation into SQL |
| T-03-02 | DoS (unbounded result) | read_ticks on very large date range | accept | Local backtest tool, operator-run; memory use bounded by operator's chosen range. Pagination can be added in Phase 4+ if needed. |
| T-03-03 | Info Disclosure | error messages leak DB internals | mitigate | `anyhow::Context` wraps errors with safe string labels; raw sqlx errors propagated only in local CLI logs |
| T-03-04 | Integrity (bad parse) | NUMERIC→u128 conversion on corrupted row | mitigate | `String::parse::<u128>()` with `.context()` returns error instead of panic; row rejected if not representable |
</threat_model>
