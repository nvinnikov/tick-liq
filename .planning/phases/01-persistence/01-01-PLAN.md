    ```rust
    use anyhow::{Context, Result};
    use chrono::{DateTime, Utc};
    use sqlx_core::executor::Executor;
    use sqlx_core::query::query;
    use sqlx_postgres::PgPool;

    #[derive(Debug, Clone)]
    pub struct PoolTick {
        pub pool_address: String,
        pub slot: i64,
        pub tick_current: i32,
        pub sqrt_price: u128,
        pub liquidity: u128,
        pub fee_growth_global_a: u128,
        pub fee_growth_global_b: u128,
        pub observed_at: DateTime<Utc>,
    }

    pub async fn write_pool_tick(pool: &PgPool, tick: &PoolTick) -> Result<()> {
        // u128 -> decimal string; NUMERIC(80,0) accepts text cast.
        let sp = tick.sqrt_price.to_string();
        let liq = tick.liquidity.to_string();
        let fga = tick.fee_growth_global_a.to_string();
        let fgb = tick.fee_growth_global_b.to_string();

        pool.execute(
            query(
                "INSERT INTO pool_ticks \
                 (time, pool_address, slot, tick_current, sqrt_price, liquidity, fee_growth_global_a, fee_growth_global_b) \
                 VALUES ($1, $2, $3, $4, $5::numeric, $6::numeric, $7::numeric, $8::numeric) \
                 ON CONFLICT (pool_address, slot) DO NOTHING",
            )
            .bind(tick.observed_at)
            .bind(&tick.pool_address)
            .bind(tick.slot)
            .bind(tick.tick_current)
            .bind(sp)
            .bind(liq)
            .bind(fga)
            .bind(fgb),
        )
        .await
        .context("write_pool_tick failed")?;
        Ok(())
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| WebSocket feed → writer | Untrusted pool state values enter the process and are written to DB |
| Writer → Postgres | Internal, but SQL injection surface if bind params bypassed |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Tampering | write_pool_tick SQL | mitigate | All values passed via `.bind()`; no string interpolation into SQL |
| T-01-02 | Denial of Service | pool_ticks unbounded growth | accept | TimescaleDB retention policy deferred to ops phase; low risk in shadow window |
| T-01-03 | Information Disclosure | DATABASE_URL in error messages | mitigate | `connect` uses `with_context` with URL — acceptable for dev; flag for prod config review |
| T-01-04 | Repudiation | Duplicate slot writes | mitigate | UNIQUE (pool_address, slot) + ON CONFLICT DO NOTHING ensures idempotency |
</threat_model>
